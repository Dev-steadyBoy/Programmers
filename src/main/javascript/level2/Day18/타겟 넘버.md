function solution(numbers, target) {
  const len = numbers.length;
  let answer = 0;

  const dfs = (cnt, sum) => {
    if (len === cnt) {
      if (sum === target) {
        answer++;
      }
      return;
    }

    dfs(cnt + 1, sum + numbers[cnt]);
    dfs(cnt + 1, sum - numbers[cnt]);
  };

  dfs(0, 0);
  return answer;
}
