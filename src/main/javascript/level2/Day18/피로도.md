function solution(k, dungeons) {
  let clearDg = 0;
  const visited = Array.from({ length: dungeons.length }, () => false);
  console.log(visited);

  const DFS = (hp, step) => {
    for (let i = 0; i < dungeons.length; i++) {
      if (!visited[i] && hp >= dungeons[i][0]) {
        visited[i] = true;

        DFS(hp - dungeons[i][1], step + 1);
        visited[i] = false;
      }
      clearDg = Math.max(clearDg, step);
    }
  };

  DFS(k, 0);
  return clearDg;
}
