function gcd(a, b) {
  const remainder = a % b;
  if (remainder === 0) return b;
  return gcd(b, remainder);
}

function getLcm(a, b) {
  let lcm = 1;
  while (true) {
    if (lcm % a === 0 && lcm % b === 0) {
      break;
    }
    lcm++;
  }
  return lcm;
}

function solution(n, m) {
  var answer = [];
  answer.push(gcd(n, m));
  answer.push(getLcm(n, m));
  return answer;
}
