function solution(board, k) {
  var numResult = 0;
  const x = board.length;
  const y = board[0].length;
  for (let i = 0; i < x; i++) {
    if (i > k) break;

    for (let j = 0; j < y; j++) {
      if (j > k || i + j > k) break;
      numResult += board[i][j];
    }
  }
  return numResult;
}
