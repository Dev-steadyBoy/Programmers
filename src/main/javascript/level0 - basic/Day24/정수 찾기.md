function solution(num_list, n) {
  var result = num_list.sort((a, b) => a - b);
  let idx = Math.floor(result.length / 2);

  if (result[idx] === n) return 1;
  if (result[idx] > n) {
    while (true) {
      idx--;
      if (result[idx] === n) return 1;
      if (result[idx] < n || idx === -1) return 0;
    }
  } else {
    while (true) {
      idx++;
      if (result[idx] === n) return 1;
      if (result[idx] > n || idx === result.length) return 0;
    }
  }
}
