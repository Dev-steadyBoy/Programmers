```javascript
function solution(l, r) {
  let result = [];

  for (let i = l; i <= r; i++) {
    const numStr = i.toString();
    if (numStr.match(/^[05]+$/)) {
      result.push(i);
    }
  }

  if (result.length === 0) {
    result.push(-1);
  }

  return result;
}
```
