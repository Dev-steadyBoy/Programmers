function solution(num_list) {
  const plusPow = num_list.reduce((acc, cur) => acc + cur, 0);

function solution(num_list) {
  let accMul = 1;
  let accSum = 0;
  for (const num of num_list) {
    accMul *= num;
    accSum += num;
  }
  return accMul < accSum ** 2 ? 1 : 0;
}
